// 3. SECTION SWITCHER FUNCTION (Global)
function showSection(sectionId, linkElement) {
    // A. Hide all sections
    document.querySelectorAll('.content-section').forEach(sec => {
        sec.classList.add('d-none');
    });

    // B. Show target section
    document.getElementById(`${sectionId}-section`).classList.remove('d-none');

    // C. Update Sidebar Active State
    document.querySelectorAll('#sidebarNav .nav-link').forEach(nav => {
        nav.classList.remove('active');
    });
    linkElement.classList.add('active');

    // D. Update Header Title
    const titles = {
        'dashboard': 'Overview',
        'registration': 'Registration & Targets',
        'merchant': 'Merchant Management',
        'earnings': 'Wallet & Commissions',
        'reports': 'Issue Tracker & Reports'
    };
    document.getElementById('pageTitle').textContent = titles[sectionId];
}
