    const startScanBtn = document.getElementById('startEnrollmentBtn');
    if(startScanBtn) {
        startScanBtn.addEventListener('click', () => {
            const scanArea = document.getElementById('fingerprint-scan-area');
            const icon = document.getElementById('scan-icon');
            const msg = document.getElementById('scan-message');
            const prog = document.getElementById('scan-progress-wrapper');
            
            scanArea.classList.add('scanning');
            startScanBtn.disabled = true;
            msg.textContent = "Scanning... Hold still.";
            prog.classList.remove('d-none');
            document.getElementById('enrollment-progress').style.transition = "width 2s linear";
            setTimeout(() => document.getElementById('enrollment-progress').style.width = "100%", 100);
            
            setTimeout(() => {
                scanArea.classList.remove('scanning');
                icon.classList.replace('text-secondary', 'text-success');
                msg.textContent = "Success!";
                msg.classList.add('text-success');
                startScanBtn.textContent = "Save";
                startScanBtn.disabled = false;
                startScanBtn.classList.replace('btn-primary', 'btn-success');
            }, 2200);
        });
    }

    // --- 13. PANIC MODE TOGGLE ---
    document.querySelectorAll('.panic-toggle').forEach(t => {
        t.addEventListener('change', (e) => {
            const row = e.target.closest('tr');
            const inputs = row.querySelectorAll('select, input[type="number"]');
            if(e.target.checked) {
                row.classList.add('bg-danger', 'bg-opacity-10');
                inputs.forEach(i => i.disabled = true);
            } else {
                row.classList.remove('bg-danger', 'bg-opacity-10');
                inputs.forEach(i => i.disabled = false);
            }
        });
    });

    // --- 14. SETTINGS TABS ---
    window.switchTab = (tabName, event) => {
        if(event) event.preventDefault();
        document.querySelectorAll('.settings-tab').forEach(el => el.classList.add('d-none'));
        document.getElementById(`tab-${tabName}`).classList.remove('d-none');
        document.querySelectorAll('.list-group-item').forEach(btn => btn.classList.remove('active'));
        if(event) event.target.closest('.list-group-item').classList.add('active');
    };
